  let scores = [];

  choices.forEach((choice, idx) => {
    let score = choice - 4;
    let type = survey[idx].split("")[score < 0 ? 0 : 1];
    scores[type] ? (scores[type] += Math.abs(score)) : (scores[type] = Math.abs(score));
  });

}
