const solution = (num, total) => {
    // 가장 작은 수와 가장 큰 수의 평균이 전체 평균과 동일(등차수열 특징 이용)
    // 가장 작은 수 찾기
    let min = (total / num) + ((1 - num) / 2)
    // min부터 시작하고 길이가 num인 배열 반환
    return Array.from({length:num}, (_, i) => i + min)
}
