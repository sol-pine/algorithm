function solution(s) {
  if (s.length === 0) return 0;
  
  let x = s[0];
  let i = 0;
  let xCnt = 0;
  let cnt = 0;

  for (i = 0; i < s.length; i++) {
    x === s[i] ? xCnt++ : cnt++;
    if (xCnt === cnt) break;
  }
  
  if (xCnt !== cnt) return 1;
  
  return 1 + solution(s.slice(i + 1, s.length));
}
