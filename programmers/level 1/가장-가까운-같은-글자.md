function solution(s) {
  let answer = [];
  let prev = [];

  for (let i = 0; i < s.length; i++) {
    if (prev.includes(s[i])) answer.push(i - prev.lastIndexOf(s[i]));
    else answer.push(-1);
    prev.push(s[i]);
  }
  return answer;
}
