function solution(board, moves) {
  let answer = 0;

  let stack = [];
  let i = 0;
  let m = 0;

  while (m < moves.length) {
    let move = moves[m] - 1;
    if (i >= board.length) {
      i = 0;
      m++;
      continue;
    }
    if (board[i][move] === 0) i++;
    else {
      let last = stack[stack.length - 1];
      if (board[i][move] === last) {
        stack.pop();
        answer += 2;
      } else stack.push(board[i][move]);
      board[i][move] = 0;
      i = 0;
      m++;
    }
  }
  return answer;
}
